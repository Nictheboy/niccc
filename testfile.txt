int fib(int i){
    if (i == 1) {
        printf("fib(%d) = %d\n", i, 1);
        return 1;
    }
    if (i == 2) {
        printf("fib(%d) = %d\n", i, 1);
        return 1;
    }
    int ret = fib(i - 1) + fib(i - 2);
    printf("fib(%d) = %d\n", i, ret);
    return ret;
}
int main () {
    printf("%d, %d\n", 1, 2);
    printf(fib(getint()));
    return 0;
}
